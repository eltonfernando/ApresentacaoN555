\subsection{equação de carga do capacitor}
% ---------------------------------------------------------------------------- %
% ---------------------------------------------------------------------------- %
% slide 33 >
% ---------------------------------------------------------------------------- %
\begin{frame} 
\frametitle{\textbf{equação de carga do capacitor}}
	\begin{block}{Divisor de tensão}
	\begin{equation}
	 V_c(J\omega)=\frac{V_{cc}\frac{1}{J\omega C}}{R_1+R_2+\frac{1}{J\omega C}}=\frac{1}{J\omega C (R_1+R_2)+1}
	\end{equation}
	\end{block}
	\pause
	\begin{block}{Se $J\omega = S $ temos:}
	\begin{equation}
	 V_c(S)=\frac{V_{cc}}{SC(R_1+R_2)+1}=V_{cc}\frac{\frac{1}{C(R_1+R_2)}}{S+ \frac{1}{C(R_1+R_2)}}
	\end{equation}
	\end{block}
	\pause
	\begin{block}{Para entrada degrau com amplitude $A$:}
	\begin{equation}
	 V_c(S)=\frac{1}{C(R_1+R_2)}\left[\frac{A}{S\left(S+\frac{1}{C(R_1+R_2)}\right)}\right]
	\end{equation}
	\end{block}
\end{frame}
% ---------------------------------------------------------------------------- %
% slide 35 >
% ---------------------------------------------------------------------------- %
\begin{frame} 
\frametitle{\textbf{equação de carga do capacitor}}
	\begin{block}{}
	 \begin{equation}
	 V_c(S)=\frac{1}{C(R_1+R_2)}\left[\frac{A}{S\left(S+\frac{1}{C(R_1+R_2)}\right)}\right]
	\end{equation}
	\end{block}
	\pause
	\begin{block}{Tabela Laplace:}
	\begin{equation}
	 \mathscr{L} \left\{1-e^{-at}\right\}  =   \frac{a}{s(s+a)}
	\end{equation}
	\end{block}
	\pause
	\begin{block}{Resposta no tempo.}
	\begin{equation}
	\label{eq:carga}
	 v_c(t)=A\left\{1-e^{-\left(\frac{1}{C(R_1+R_2)}\right)t}\right\}
	\end{equation}
	\end{block}
\end{frame}
